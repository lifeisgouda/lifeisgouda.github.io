insert()

  • 값을 삽입
  • 기존 값 있을 경우 덮어쓰기
numbers = [1, 2, 3, 4, 5]
numbers.insert(0, 0)    # 0번째에 0 넣음
print(numbers)
# [0, 1, 2, 3, 4, 5]

extend()

  • 더하기 연산자와의 차이 생각
  • 기존 변수에 확장됨. 값 변함.
numbers = [1, 2, 3, 4, 5]
numbers2 = [6, 7, 8]
numbers.extend(numbers2)
pirnt(numbers)
# [1, 2, 3, 4, 5, 6, 7, 8]
